1.1 Multi-Tool Server Pattern | 多工具伺服器模式 | マルチツールサーバーパターン

// Production pattern: Tool registry with dynamic routing
// 正式環境模式：工具註冊表與動態路由
// 本番パターン：ツールレジストリと動的ルーティング
const TOOL_REGISTRY = new Map();

function registerTool(name, schema, handler) {
  TOOL_REGISTRY.set(name, { schema, handler });
}

// Register tools | 註冊工具 | ツールを登録
registerTool('search', {
  description: 'Search the knowledge base',
  inputSchema: {
    type: 'object',
    properties: {
      query: { type: 'string', description: 'Search query' },
      filters: { type: 'object', description: 'Optional filters' },
    },
    required: ['query'],
  },
}, async (args, env) => {
  const results = await env.DB.prepare(
    `SELECT * FROM knowledge WHERE content LIKE ? LIMIT 10`
  ).bind(`%${args.query}%`).all();
  return results.results;
});

registerTool('analyze', {
  description: 'Analyze data with AI',
  inputSchema: {
    type: 'object',
    properties: {
      data: { type: 'string', description: 'Data to analyze' },
      perspective: { type: 'string', enum: ['technical', 'business', 'security'] },
    },
    required: ['data'],
  },
}, async (args, env) => {
  // Call free LLM for analysis
  return await callLLM(env, args.data, args.perspective || 'technical');
});

// Dynamic tool call routing | 動態工具呼叫路由 | 動的ツール呼び出しルーティング
async function handleToolCall(params, env, rateLimit) {
  const tool = TOOL_REGISTRY.get(params.name);
  if (!tool) throw new Error(`Unknown tool: ${params.name}`);

  const result = await tool.handler(params.arguments, env);
  return { content: [{ type: 'text', text: JSON.stringify(result, null, 2) }] };
}

1.2 Authentication Layer | 驗證層 | 認証レイヤー

// Three-tier auth: Public → Free → Pro
// 三層驗證：匿名 → 免費 → 專業版
// 3段階認証：匿名 → 無料 → プロ
async function authenticate(request, env) {
  const apiKey = request.headers.get('X-API-Key') || '';

  if (!apiKey) {
    return { tier: 'anonymous', limit: 1, features: ['basic'] };
  }

  const proData = await env.KV.get(`prokey:${apiKey}`, { type: 'json' });
  if (proData) {
    return {
      tier: 'pro',
      limit: proData.calls_per_day || 1000,
      features: ['basic', 'advanced', 'export', 'priority'],
      email: proData.email,
    };
  }

  // Check if it's a valid free-tier key
  const freeData = await env.KV.get(`freekey:${apiKey}`, { type: 'json' });
  if (freeData) {
    return { tier: 'free', limit: 10, features: ['basic'] };
  }

  return { tier: 'invalid', limit: 0, features: [] };
}

3.1 Payment Integration (Stripe) | 付款整合 | 決済連携（Stripe）

// Stripe webhook handler | Stripe Webhook 處理器 | Stripe Webhookハンドラー
async function handleStripeWebhook(request, env) {
  const signature = request.headers.get('stripe-signature');
  const body = await request.text();

  // Verify signature | 驗證簽名 | 署名を検証
  const isValid = await verifyStripeSignature(body, signature, env.STRIPE_WEBHOOK_SECRET);
  if (!isValid) return new Response('Invalid signature', { status: 401 });

  const event = JSON.parse(body);

  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;
    const email = session.customer_email;

    // Generate Pro API key | 生成專業版 API 金鑰 | プロAPIキーを生成
    const apiKey = 'pro_' + crypto.randomUUID().replace(/-/g, '');
    await env.KV.put(`prokey:${apiKey}`, JSON.stringify({
      email, plan: 'pro', created: new Date().toISOString(),
      calls_per_day: 1000,
    }), { expirationTtl: 365 * 86400 });

    // Send key via email | 透過電子郵件發送金鑰 | メールでキーを送信
    await sendEmail(env, email, 'Your Pro API Key', `Your key: ${apiKey}`);
  }

  return new Response('OK');
}
