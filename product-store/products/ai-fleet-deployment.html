2.1 Health Commander — Self-Healing Pattern

// health-commander/worker.js
const ENDPOINTS = [
  { name: 'orchestrator', url: 'https://my-orchestrator.YOUR.workers.dev/health' },
  { name: 'intel-ops', url: 'https://my-intel-ops.YOUR.workers.dev/health' },
  { name: 'executor', url: 'https://my-cloud-executor.YOUR.workers.dev/health' },
  { name: 'revenue', url: 'https://my-revenue-sentinel.YOUR.workers.dev/health' },
  { name: 'content', url: 'https://my-content-engine.YOUR.workers.dev/health' },
];

export default {
  async scheduled(event, env) {
    const results = [];

    for (const ep of ENDPOINTS) {
      const start = Date.now();
      try {
        const resp = await fetch(ep.url, { signal: AbortSignal.timeout(5000) });
        const latency = Date.now() - start;
        const ok = resp.status === 200;

        results.push({ name: ep.name, ok, latency, status: resp.status });

        if (!ok) {
          // Auto-restart attempt (3 retries)
          for (let i = 0; i < 3; i++) {
            await new Promise(r => setTimeout(r, 2000));
            const retry = await fetch(ep.url, { signal: AbortSignal.timeout(5000) });
            if (retry.status === 200) { results.at(-1).recovered = true; break; }
          }
        }

        // Store SLA data
        await env.DB.prepare(
          'INSERT INTO fleet_events (worker_id, event_type, severity, data) VALUES (?,?,?,?)'
        ).bind(ep.name, 'health_check', ok ? 'info' : 'critical',
          JSON.stringify({ latency, status: resp.status })
        ).run();
      } catch (e) {
        results.push({ name: ep.name, ok: false, error: e.message });
      }
    }

    // Alert on failures
    const failures = results.filter(r => !r.ok);
    if (failures.length > 0) {
      await sendTelegram(env,
        `🚨 Fleet Health Alert\n${failures.map(f => `❌ ${f.name}: ${f.error || f.status}`).join('\n')}`
      );
    }
  },
};

2.2 Revenue Sentinel — Order Detection

// revenue-sentinel/worker.js
export default {
  async scheduled(event, env) {
    // Check Stripe for new payments
    const resp = await fetch('https://api.stripe.com/v1/charges?limit=10', {
      headers: { 'Authorization': `Bearer ${env.STRIPE_SECRET_KEY}` },
    });
    const charges = await resp.json();

    for (const charge of charges.data) {
      // Check if already recorded
      const existing = await env.DB.prepare(
        'SELECT id FROM revenue_ledger WHERE stripe_charge_id = ?'
      ).bind(charge.id).first();

      if (!existing && charge.paid) {
        await env.DB.prepare(
          'INSERT INTO revenue_ledger (stripe_charge_id, amount, currency, product, created_at) VALUES (?,?,?,?,?)'
        ).bind(charge.id, charge.amount / 100, charge.currency,
          charge.metadata?.product_id || 'unknown', charge.created
        ).run();

        await sendTelegram(env,
          `💰 New Order: $${charge.amount / 100} — ${charge.metadata?.product_id || 'unknown'}`
        );
      }
    }

    // Daily summary (at midnight UTC)
    if (new Date().getHours() === 0) {
      const today = new Date().toISOString().slice(0, 10);
      const daily = await env.DB.prepare(
        'SELECT SUM(amount) as total, COUNT(*) as orders FROM revenue_ledger WHERE created_at LIKE ?'
      ).bind(`${today}%`).first();

      await sendTelegram(env,
        `📊 Daily Revenue: $${daily.total || 0} (${daily.orders || 0} orders)`
      );
    }
  },
};

3. Database Schema

-- schema.sql — Complete fleet database
CREATE TABLE fleet_tasks (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  type TEXT NOT NULL,
  priority INTEGER DEFAULT 5,
  payload TEXT,
  status TEXT DEFAULT 'pending',
  worker_id TEXT,
  result TEXT,
  created_at TEXT DEFAULT (datetime('now')),
  completed_at TEXT
);

CREATE TABLE fleet_events (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  worker_id TEXT NOT NULL,
  event_type TEXT NOT NULL,
  severity TEXT DEFAULT 'info',
  data TEXT,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE TABLE intel_feed (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  source TEXT NOT NULL,
  title TEXT,
  url TEXT,
  score INTEGER DEFAULT 0,
  raw TEXT,
  analyzed INTEGER DEFAULT 0,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE TABLE revenue_ledger (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  stripe_charge_id TEXT UNIQUE,
  amount REAL NOT NULL,
  currency TEXT DEFAULT 'usd',
  product TEXT,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE TABLE fleet_metrics (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  worker_id TEXT NOT NULL,
  metric TEXT NOT NULL,
  value REAL NOT NULL,
  created_at TEXT DEFAULT (datetime('now'))
);

-- Indexes for performance
CREATE INDEX idx_tasks_status ON fleet_tasks(status, priority DESC);
CREATE INDEX idx_events_worker ON fleet_events(worker_id, created_at);
CREATE INDEX idx_intel_date ON intel_feed(created_at);
CREATE INDEX idx_revenue_date ON revenue_ledger(created_at);

4. Fleet Dashboard API

// Add to any worker — /dashboard endpoint
async function handleDashboard(env) {
  const [tasks, events, revenue] = await Promise.all([
    env.DB.prepare(`
      SELECT status, COUNT(*) as count FROM fleet_tasks
      WHERE created_at > datetime('now', '-24 hours') GROUP BY status
    `).all(),
    env.DB.prepare(`
      SELECT worker_id, severity, COUNT(*) as count FROM fleet_events
      WHERE created_at > datetime('now', '-24 hours') GROUP BY worker_id, severity
    `).all(),
    env.DB.prepare(`
      SELECT SUM(amount) as total, COUNT(*) as orders FROM revenue_ledger
      WHERE created_at > datetime('now', '-24 hours')
    `).first(),
  ]);

  return Response.json({
    tasks: tasks.results,
    health: events.results,
    revenue: { total: revenue?.total || 0, orders: revenue?.orders || 0 },
    timestamp: new Date().toISOString(),
  });
}
