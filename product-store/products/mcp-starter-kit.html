// worker.js — Production-ready MCP Server Template
// MCP Protocol: 2025-03-26 | JSON-RPC 2.0

const SERVER_INFO = {
  name: 'your-mcp-server',
  version: '1.0.0',
  description: 'Your MCP server description',
};

const TOOLS = [
  {
    name: 'your_tool_name',
    description: 'What this tool does — be specific for AI discovery',
    inputSchema: {
      type: 'object',
      properties: {
        query: { type: 'string', description: 'The search query' },
        limit: { type: 'number', description: 'Max results (default: 10)' },
      },
      required: ['query'],
    },
  },
];

export default {
  async fetch(request, env) {
    const url = new URL(request.url);

    // Landing page
    if (url.pathname === '/' && request.method === 'GET') {
      return new Response(LANDING_HTML, {
        headers: { 'Content-Type': 'text/html; charset=utf-8' },
      });
    }

    // Health check
    if (url.pathname === '/health') {
      return Response.json({ ok: true, version: SERVER_INFO.version });
    }

    // MCP endpoint (JSON-RPC 2.0)
    if (request.method !== 'POST') {
      return Response.json({ error: 'Method not allowed' }, { status: 405 });
    }

    // Rate limiting
    const rateLimitResult = await checkRateLimit(request, env);
    if (!rateLimitResult.ok) {
      return Response.json({
        jsonrpc: '2.0',
        error: { code: -32000, message: rateLimitResult.message },
        id: null,
      }, { status: 429 });
    }

    const body = await request.json();
    const { method, params, id } = body;

    try {
      let result;
      switch (method) {
        case 'initialize':
          result = {
            protocolVersion: '2025-03-26',
            capabilities: { tools: { listChanged: false } },
            serverInfo: SERVER_INFO,
          };
          break;

        case 'tools/list':
          result = { tools: TOOLS };
          break;

        case 'tools/call':
          result = await handleToolCall(params, env, rateLimitResult);
          break;

        default:
          return Response.json({
            jsonrpc: '2.0',
            error: { code: -32601, message: `Unknown method: ${method}` },
            id,
          });
      }

      return Response.json({ jsonrpc: '2.0', result, id });
    } catch (error) {
      return Response.json({
        jsonrpc: '2.0',
        error: { code: -32603, message: error.message },
        id,
      }, { status: 500 });
    }
  },
};

// Tool execution
async function handleToolCall(params, env, rateLimit) {
  const { name, arguments: args } = params;

  switch (name) {
    case 'your_tool_name': {
      // YOUR TOOL LOGIC HERE
      const results = await yourToolFunction(args.query, args.limit || 10, env);
      return {
        content: [{
          type: 'text',
          text: JSON.stringify(results, null, 2),
        }],
      };
    }

    default:
      throw new Error(`Unknown tool: ${name}`);
  }
}

// Rate limiting (Pro tier: 1000/day, Free: 3/day)
async function checkRateLimit(request, env) {
  const apiKey = request.headers.get('X-API-Key') || '';
  const isPro = apiKey && await env.KV?.get(`prokey:${apiKey}`);
  const limit = isPro ? 1000 : 3;
  const tier = isPro ? 'pro' : 'free';

  const ip = request.headers.get('CF-Connecting-IP') || 'unknown';
  const key = isPro ? `rate:pro:${apiKey}` : `rate:free:${ip}`;
  const today = new Date().toISOString().slice(0, 10);
  const rateKey = `${key}:${today}`;

  const current = parseInt(await env.KV?.get(rateKey) || '0');
  if (current >= limit) {
    return {
      ok: false,
      message: isPro
        ? `Pro rate limit: ${limit}/day reached. Resets at midnight UTC.`
        : `Free tier: ${limit} calls/day. Upgrade: YOUR_STORE_URL`,
    };
  }

  await env.KV?.put(rateKey, String(current + 1), { expirationTtl: 86400 });
  return { ok: true, tier, remaining: limit - current - 1 };
}

CREATE TABLE IF NOT EXISTS data (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  key TEXT NOT NULL UNIQUE,
  value TEXT NOT NULL,
  metadata TEXT,
  created_at TEXT DEFAULT (datetime('now')),
  updated_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX idx_data_key ON data(key);
CREATE INDEX idx_data_created ON data(created_at);

-- Usage tracking
CREATE TABLE IF NOT EXISTS usage_log (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  api_key TEXT,
  tool_name TEXT NOT NULL,
  ip_hash TEXT,
  tier TEXT DEFAULT 'free',
  created_at TEXT DEFAULT (datetime('now'))
);

6. Deploy in 30 Seconds

# 1. Install dependencies
npm install

# 2. Create D1 database
npx wrangler d1 create your-mcp-db
# Copy database_id to wrangler.toml

# 3. Create KV namespace
npx wrangler kv namespace create KV
# Copy id to wrangler.toml

# 4. Run schema
npx wrangler d1 execute your-mcp-db --file schema.sql

# 5. Deploy!
npx wrangler deploy
