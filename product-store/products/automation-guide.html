Every AI agent follows a simple loop:

while (running) {
  // 1. OBSERVE — Gather information
  const data = await collectData(sources);

  // 2. THINK — Analyze with AI
  const analysis = await llm.analyze(data, context);

  // 3. ACT — Execute decisions

2.2 Building a Market Intel Agent | 建立市場情報代理 | マーケットインテルエージェントの構築

Let's build a real agent that monitors HackerNews for AI trends:

// agents/intel-agent.js
const SOURCES = {
  hn: 'https://hacker-news.firebaseio.com/v0',
};

async function collectHNTrends() {
  // Fetch top stories
  const topIds = await fetch(`${SOURCES.hn}/topstories.json`)
    .then(r => r.json());

  // Get details for top 30
  const stories = await Promise.all(
    topIds.slice(0, 30).map(id =>
      fetch(`${SOURCES.hn}/item/${id}.json`).then(r => r.json())
    )
  );

  // Filter AI-related
  const aiStories = stories.filter(s =>
    /\b(ai|llm|gpt|claude|agent|mcp)\b/i.test(s.title)
  );

  return aiStories.map(s => ({
    title: s.title,
    url: s.url,
    score: s.score,
    comments: s.descendants || 0,
  }));
}

// Analyze with any free LLM API
async function analyzeWithAI(stories) {
  const prompt = `Analyze these AI news items and identify:
1. Top 3 trends
2. Business opportunities
3. Threats to watch

Stories: ${JSON.stringify(stories, null, 2)}`;

  // Using free Groq API (llama-3.3-70b)
  const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${process.env.GROQ_API_KEY}`,
    },
    body: JSON.stringify({
      model: 'llama-3.3-70b-versatile',
      messages: [{ role: 'user', content: prompt }],
      max_tokens: 1000,
    }),
  });

  return response.json();
}

# Format: minute hour day-of-month month day-of-week

// worker.js
export default {
  async scheduled(event, env, ctx) {
    // This runs on schedule — no HTTP request needed
    const intel = await collectHNTrends();
    const analysis = await analyzeWithAI(intel);

    // Store results in D1
    await env.DB.prepare(
      'INSERT INTO intel_reports (data, created_at) VALUES (?, ?)'
    ).bind(JSON.stringify(analysis), new Date().toISOString()).run();

    // Alert via Telegram if high-value intel found
    if (analysis.highPriority) {
      await sendTelegram(env, `High-value intel detected:\n${analysis.summary}`);
    }
  },
};

async function sendTelegram(env, text) {
  const url = `https://api.telegram.org/bot${env.TELEGRAM_BOT_TOKEN}/sendMessage`;
  await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      chat_id: env.TELEGRAM_CHAT_ID,
      text: text,
      parse_mode: 'Markdown',
    }),
  });
}

// Handle incoming Telegram webhook
async function handleTelegramWebhook(request, env) {
  const update = await request.json();
  const message = update.message;
  if (!message?.text) return new Response('OK');

  const chatId = message.chat.id;
  const text = message.text.trim();

  // Simple command router
  const commands = {
    '/status': () => getSystemStatus(env),
    '/intel': () => getLatestIntel(env),
    '/report': () => generateReport(env),
    '/help': () => 'Commands: /status, /intel, /report',
  };

  const handler = commands[text.split(' ')[0]];
  const reply = handler ? await handler() : 'Unknown command. Try /help';

  await sendTelegram(env, reply);
  return new Response('OK');
}

6.1 Health Check Pattern | 健康檢查模式 | ヘルスチェックパターン

// Every worker should have a /health endpoint
if (url.pathname === '/health') {
  const checks = {
    worker: 'ok',
    d1: 'unknown',
    kv: 'unknown',
    uptime: Date.now() - startTime,
  };

  try {
    await env.DB.prepare('SELECT 1').run();
    checks.d1 = 'ok';
  } catch (e) {
    checks.d1 = `error: ${e.message}`;
  }

  try {
    await env.KV.put('health-check', Date.now().toString());
    checks.kv = 'ok';
  } catch (e) {
    checks.kv = `error: ${e.message}`;
  }

  const allOk = checks.d1 === 'ok' && checks.kv === 'ok';
  return Response.json(checks, { status: allOk ? 200 : 503 });
}

6.2 Self-Healing Patterns | 自我修復模式 | 自己修復パターン

// Retry with exponential backoff
async function retryWithBackoff(fn, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      const delay = Math.min(1000 * Math.pow(2, i), 10000);
      await new Promise(r => setTimeout(r, delay));
    }
  }
}

// Circuit breaker
class CircuitBreaker {
  constructor(threshold = 3, resetMs = 60000) {
    this.failures = 0;
    this.threshold = threshold;
    this.resetMs = resetMs;
    this.openUntil = 0;
  }

  async execute(fn) {
    if (Date.now() < this.openUntil) {
      throw new Error('Circuit breaker OPEN');
    }
    try {
      const result = await fn();
      this.failures = 0;
      return result;
    } catch (error) {
      this.failures++;
      if (this.failures >= this.threshold) {
        this.openUntil = Date.now() + this.resetMs;
      }
      throw error;
    }
  }
}
