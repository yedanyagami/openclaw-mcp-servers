1. The OODA Loop Explained

1.1 OBSERVE вҖ” Intel Collection

Gather intelligence from free APIs every 30 minutes:

// yedan-intel-ops/worker.js (simplified)
const INTEL_SOURCES = [
  {
    name: 'hackernews',
    url: 'https://hacker-news.firebaseio.com/v0/topstories.json',
    parser: parseHN,
    interval: 30, // minutes
  },
  {
    name: 'github-trending',
    url: 'https://api.github.com/search/repositories?q=mcp+server&sort=stars&order=desc',
    parser: parseGitHub,
    interval: 60,
  },
  {
    name: 'npm-new',
    url: 'https://registry.npmjs.org/-/v1/search?text=mcp+server&size=20',
    parser: parseNPM,
    interval: 120,
  },
];

async function observe(env) {
  const intel = [];
  for (const source of INTEL_SOURCES) {
    try {
      const resp = await fetch(source.url, { signal: AbortSignal.timeout(5000) });
      const data = await resp.json();
      intel.push(...source.parser(data));
    } catch (e) {
      // Log but don't fail вҖ” resilience
      console.error(`${source.name} failed: ${e.message}`);
    }
  }

  // Store raw intel
  const batch = intel.map(item =>
    env.DB.prepare('INSERT INTO intel_feed (source, title, url, score, raw) VALUES (?,?,?,?,?)')
      .bind(item.source, item.title, item.url, item.score, JSON.stringify(item))
  );
  await env.DB.batch(batch);

  return intel;
}

1.3 DECIDE вҖ” Task Creation

async function decide(env, scoredItems) {
  const tasks = [];

  for (const item of scoredItems) {
    const effectiveScore = Math.round(item.opportunity_score * item.confidence);

    // Skip low-value or low-confidence items
    if (effectiveScore < 5) continue;

    // Check failure patterns (circuit breaker)
    const failCount = parseInt(await env.KV.get(`fail:${item.type}`) || '0');
    if (failCount >= 3) continue; // Skip failing task types for 1 hour

    tasks.push({
      type: mapToTaskType(item),
      priority: effectiveScore,
      payload: item,
      source: 'orchestrator-decide',
    });
  }

  // Store tasks in D1
  for (const task of tasks.sort((a, b) => b.priority - a.priority)) {
    await env.DB.prepare(
      'INSERT INTO fleet_tasks (type, priority, payload, status) VALUES (?,?,?,?)'
    ).bind(task.type, task.priority, JSON.stringify(task.payload), 'pending').run();
  }

  return tasks;
}

1.4 ACT вҖ” Task Execution

// yedan-cloud-executor/worker.js (simplified)
async function execute(env) {
  // Pull highest-priority pending task
  const task = await env.DB.prepare(
    'SELECT * FROM fleet_tasks WHERE status = ? ORDER BY priority DESC LIMIT 1'
  ).bind('pending').first();

  if (!task) return { status: 'idle' };

  // Mark as in-progress
  await env.DB.prepare('UPDATE fleet_tasks SET status = ?, worker_id = ? WHERE id = ?')
    .bind('in_progress', 'cloud-executor', task.id).run();

  try {
    const result = await executeTask(env, task);

    // Mark complete
    await env.DB.prepare('UPDATE fleet_tasks SET status = ?, result = ? WHERE id = ?')
      .bind('completed', JSON.stringify(result), task.id).run();

    // Record lesson for self-evolution
    await recordLesson(env, task, result, true);

    return result;
  } catch (error) {
    await env.DB.prepare('UPDATE fleet_tasks SET status = ?, result = ? WHERE id = ?')
      .bind('failed', error.message, task.id).run();

    await recordLesson(env, task, { error: error.message }, false);
    throw error;
  }
}

1.5 REFLECT вҖ” Self-Evolution (v3.0)

async function reflect(env) {
  // Analyze completed and failed tasks from last cycle
  const recent = await env.DB.prepare(`
    SELECT type, status, COUNT(*) as count,
      AVG(CASE WHEN status='completed' THEN 1.0 ELSE 0.0 END) as success_rate
    FROM fleet_tasks
    WHERE created_at > datetime('now', '-6 hours')
    GROUP BY type, status
  `).all();

  // Update failure patterns
  for (const row of recent.results) {
    if (row.status === 'failed' && row.count >= 3) {
      await env.KV.put(`fail:${row.type}`, String(row.count), { expirationTtl: 3600 });
    }
  }

  // Write feedback to Graph RAG for cross-agent learning
  await writeReflectionToGraphRAG(env, recent.results);

  return { reflected: recent.results.length, patterns: recent.results };
}

2. D1 Database Schema

-- Full schema for OODA fleet
CREATE TABLE fleet_tasks (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  type TEXT NOT NULL,
  priority INTEGER DEFAULT 5,
  payload TEXT,
  status TEXT DEFAULT 'pending',
  worker_id TEXT,
  result TEXT,
  created_at TEXT DEFAULT (datetime('now')),
  completed_at TEXT
);

CREATE TABLE intel_feed (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  source TEXT NOT NULL,
  title TEXT NOT NULL,
  url TEXT,
  score INTEGER DEFAULT 0,
  raw TEXT,
  analyzed INTEGER DEFAULT 0,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE TABLE fleet_events (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  worker_id TEXT NOT NULL,
  event_type TEXT NOT NULL,
  severity TEXT DEFAULT 'info',
  data TEXT,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE TABLE agent_lessons (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  agent_id TEXT NOT NULL,
  situation TEXT NOT NULL,
  action_taken TEXT,
  outcome TEXT,
  score INTEGER DEFAULT 0,
  created_at TEXT DEFAULT (datetime('now'))
);

CREATE INDEX idx_tasks_status ON fleet_tasks(status, priority DESC);
CREATE INDEX idx_intel_source ON intel_feed(source, created_at);
CREATE INDEX idx_events_worker ON fleet_events(worker_id, created_at);
CREATE INDEX idx_lessons_agent ON agent_lessons(agent_id, created_at);
