1.1 Pre-Commit Security Check

// .claude/hooks/pre-commit-security.js
// Prevents committing secrets, large files, or debug code
export default {
  event: 'pre_commit',
  async handler({ files }) {
    const issues = [];

    for (const file of files) {
      const content = await readFile(file.path);

      // Check for hardcoded secrets
      const secretPatterns = [
        /(?:api[_-]?key|secret|token|password)\s*[:=]\s*['"][^'"]{8,}/gi,
        /sk_live_[a-zA-Z0-9]{20,}/g,           // Stripe live key
        /ghp_[a-zA-Z0-9]{36}/g,                 // GitHub PAT
        /AIza[a-zA-Z0-9_-]{35}/g,               // Google API key
        /-----BEGIN (?:RSA )?PRIVATE KEY-----/g, // Private keys
      ];

      for (const pattern of secretPatterns) {
        if (pattern.test(content)) {
          issues.push({ file: file.path, issue: 'Possible secret detected', pattern: pattern.source });
        }
      }

      // Check for debug code
      if (/console\.log|debugger|TODO:.*REMOVE/i.test(content)) {
        issues.push({ file: file.path, issue: 'Debug code detected' });
      }

      // Check file size (> 1MB)
      if (content.length > 1_000_000) {
        issues.push({ file: file.path, issue: `Large file: ${(content.length / 1_000_000).toFixed(1)}MB` });
      }
    }

    if (issues.length > 0) {
      return { blocked: true, message: `Security check failed:\n${issues.map(i => `  ${i.file}: ${i.issue}`).join('\n')}` };
    }

    return { blocked: false };
  },
};

1.2 Auto-Format on Save

// .claude/hooks/auto-format.js
export default {
  event: 'post_tool_use',
  match: { tool: ['Write', 'Edit'] },
  async handler({ tool_use }) {
    const ext = tool_use.file_path?.split('.').pop();
    const formatters = {
      js: 'npx prettier --write',
      ts: 'npx prettier --write',
      py: 'python -m black',
      go: 'gofmt -w',
      rs: 'rustfmt',
    };

    const formatter = formatters[ext];
    if (formatter) {
      await exec(`${formatter} "${tool_use.file_path}"`);
    }
  },
};

1.3 Test Runner After Code Changes

// .claude/hooks/auto-test.js
export default {
  event: 'post_tool_use',
  match: { tool: ['Write', 'Edit'] },
  async handler({ tool_use }) {
    // Only run tests for source files
    if (!/\.(js|ts|py|go)$/.test(tool_use.file_path)) return;
    if (/\.(test|spec)\.(js|ts)$/.test(tool_use.file_path)) return; // Skip test files

    // Find and run related tests
    const testFile = tool_use.file_path.replace(/\.(js|ts)$/, '.test.$1');
    const exists = await fileExists(testFile);

    if (exists) {
      const result = await exec(`npx jest ${testFile} --no-coverage 2>&1`);
      if (result.exitCode !== 0) {
        return { message: `Tests failed:\n${result.stdout.slice(-500)}` };
      }
    }
  },
};
